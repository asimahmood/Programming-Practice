    cin >> val1;
    num1 = val1;
    cout << "Enter the number 2: ";
    cin >> val2;
    num2 = val2;
    
    while (num1 != 0 && num2 != 0)
    {
        sum1 = sum1 + num1 % 10;
        num1 = num1 / 10;
        
        sum2 = sum2 + num2 % 10;
        num2 = num2 / 10;
        
    }
    if (sum1==sum2)
    {
        cout << "Yes, the sum of the digits of numbers "<<val1<<" & "<<val2<<" is same."<<endl;

    else
    {
    cout << "The sum of the numbers: "
    << val1 << " & " << val2<<" is "<< sum1<< " - "<< sum2<<endl;
    }

}

#include <iostream>
using namespace std;

int main()
{
    int number, max =0, rem;
    
    cout << "Enter a number: ";
    cin >> number;
    
    while ( number != 0)
    {
        rem = number % 10;
        if(rem>max)
        {
            max = rem;
        }
        number = number /10;
    }
    
    cout << "The largest digit in that integer is: " << max << endl;
    
    
    return 0;
}

#include <iostream>
#include <string>
#include <fstream>
using namespace std;

int main()
{
    string str;
    
    ifstream input;
    input.open ("input.txt");
    ofstream out;
    out.open ("output.txt");
    
    while (!input.eof())
    {
        input >> str;
        for (size_t i=0; i< str.length(); ++i)
        {
            str[i] = ::toupper(str[i]);
        }
        cout<<str;
        out<<str;
    } 
    cout<<endl;
    return 0;
}
